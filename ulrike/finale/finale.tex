\newcommand\tikzlingfirework[3]{%#1 tikzling, #2 x faktor, #3 y factor
   \foreach \radius in {4,...,16}
     {%
        \begin{tikzpicture}[]
        \path[use as bounding box](0,0) rectangle (\textwidth,\textheight);
         % fireworks
         \pgfmathparse{100-(\radius-1)*10}%;
         \begin{scope}[decoration={markings,mark=between positions 0 and 1 step 0.2 with{\path (0,0)pic[scale=\fpeval{0.1+0.05*\radius}]{#1};}}]    
         \path[postaction=decorate] (0.#2\textwidth,0.#3\textheight) circle (\radius ex);
         \end{scope}
        \end{tikzpicture}
        \newpage
        \begin{tikzpicture}[]
        \path[use as bounding box](0,0) rectangle (\textwidth,\textheight);
         % fireworks
         \pgfmathparse{100-(\radius-1)*10}%;
         \begin{scope}[decoration={markings,mark=between positions 0 and 1 step 0.2 with{\path (0,0)pic[scale=\fpeval{0.1+0.05*\radius}]{#1};}}]    
         \path[postaction=decorate] (0.#2\textwidth,0.#3\textheight) circle (\radius.5 ex);
         \end{scope}
        \end{tikzpicture}
        \newpage}}
