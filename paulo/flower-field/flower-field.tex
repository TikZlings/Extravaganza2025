\newcommand{\drawMultipleHops}[2]{%
    \begin{tikzpicture}[scale=1.2]
        \path (-0.5,1.5) rectangle (11.5,4);
        \def\totalwidth{10}
        \def\segments{#2}
        \pgfmathsetmacro{\parabolawidth}{\totalwidth / \segments}

        \foreach \n in {0,...,#2} {
            \pgfmathsetmacro{\a}{-1}
            \pgfmathsetmacro{\h}{\n * \parabolawidth + 0.5 * \parabolawidth}
            \pgfmathsetmacro{\k}{2}
            \pgfmathsetmacro{\startx}{\n * \parabolawidth}
            \pgfmathsetmacro{\endx}{(\n + 1) * \parabolawidth}

            \draw[domain=\startx:\endx, samples=50, draw=none]
                plot (\x, {\a * (\x - \h)^2 + \k});
        }

        \pgfmathsetmacro{\xpos}{#1}
        \pgfmathsetmacro{\nparabola}{floor(#1 / \parabolawidth)}
        \pgfmathsetmacro{\hcurrent}{\nparabola * \parabolawidth + 0.5 * \parabolawidth}
        \pgfmathsetmacro{\ypos}{-1 * (#1 - \hcurrent)^2 + 2}
        \path (\xpos, \ypos) pic{anteater};
    \end{tikzpicture}%
}
