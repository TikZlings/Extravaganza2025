\newcounter{mystep}
\newcommand{\setstep}{%

  \setcounter{mystep}{0}
\fi%
\stepcounter{mystep}%
}

% trick taken from https://topanswers.xyz/tex?q=1989
\tikzset{
    use page relative coordinates/.style={
        shift={(current page.south west)},
        x={(current page.south east)},
        y={(current page.north west)}
    },
}
